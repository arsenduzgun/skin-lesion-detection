build the image: docker build -t model-dev-image .

run the container: docker run -d --rm --name model-dev-container --shm-size=8g -p 8888:8888 -v "${PWD}:/model-dev" model-dev-image jupyter lab --ip=0.0.0.0 --port=8888 --no-browser --IdentityProvider.token=devtoken

server: http://127.0.0.1:8888/?token=devtoken

stop and delete the container: docker stop model-dev-container

delete the image: docker rmi model-dev-image
